Architecture MVC + DAO — Structure du projet
/*
 * ARCHITECTURE MVC (Modèle - Vue - Contrôleur)
 * ─────────────────────────────────────────────
 * modeles/       → Classes PHP représentant les entités (Clients, Produits…)
 * dao/           → Accès base de données via PDO (requêtes préparées)
 * controleurs/   → Logique métier, appel des DAO, transmission aux vues
 * vues/          → Affichage HTML uniquement (aucune logique)
 */

// ── Modèle : Commande.php ──────────────────────────────────────────
class Commande {
    private $id, $date, $statut, $idClient;

    public function __construct($date, $statut, $idClient, $id=null) {
        $this->id = $id;
        $this->date = $date;
        $this->statut = $statut;
        $this->idClient = $idClient;
    }
    // Getters / Setters
    public function getId()      { return $this->id; }
    public function getDate()    { return $this->date; }
    public function getStatut()  { return $this->statut; }
}

// ── DAO : CommandesDAO.php ─────────────────────────────────────────
class CommandesDAO extends BaseDAO {

    /**
     * Retourne toutes les commandes d'un client avec les lignes associées
     * @param  int   $idClient  Identifiant du client en base
     * @return array            Tableau associatif [jour, libelle, qte, prix]
     */
    public function getByClient($idClient) {
        $sql = "SELECT c.id, c.date, c.statut,
                       p.libelle, lc.qte, p.prix
                FROM   commandes c
                JOIN   ligne_commande lc ON lc.idCommande = c.id
                JOIN   produits p        ON p.id = lc.idProduit
                WHERE  c.idClient = :id
                ORDER  BY c.date DESC";
        $stmt = $this->pdo()->prepare($sql);
        $stmt->execute([':id' => $idClient]);
        return $stmt->fetchAll(PDO::FETCH_ASSOC);
    }

    /**
     * Crée une nouvelle commande en base de données
     * @param  Commande $cmd  Objet Commande à insérer
     * @return bool           true si succès
     */
    public function ajouter(Commande $cmd) {
        $sql = "INSERT INTO commandes (date, statut, idClient)
                VALUES (:date, :statut, :idClient)";
        $stmt = $this->pdo()->prepare($sql);
        return $stmt->execute([
            ':date'     => $cmd->getDate(),
            ':statut'   => $cmd->getStatut(),
            ':idClient' => $cmd->getIdClient()
        ]);
    }
}

VisiteursDAO.php — Architecture MVC + DAO commentée (Site Fâ)
// ── Modèle : Visiteur.php ─────────────────────────────────────────
class Visiteur {
    private $id, $nom, $prenom, $email, $tel;

    public function __construct($nom, $prenom, $email, $tel, $id=null) {
        $this->nom = $nom;
        $this->prenom = $prenom;
        $this->email = $email;
        $this->tel = $tel;
        $this->id = $id;
    }
    public function getNom()    { return $this->nom; }
    public function getPrenom() { return $this->prenom; }
    public function getEmail()  { return $this->email; }
    public function getTel()    { return $this->tel; }
}

// ── DAO : VisiteursDAO.php ────────────────────────────────────────
class VisiteursDAO extends BaseDAO {

    /**
     * Insère un visiteur en base de données (requête préparée)
     * @param  Visiteur $v  Objet visiteur à enregistrer
     * @return bool         true si l'insertion a réussi
     */
    public function ajouter(Visiteur $v) {
        $sql = "INSERT INTO visiteurs (nom, prenom, email, tel)
                VALUES (:nom, :prenom, :email, :tel)";
        $stmt = $this->pdo()->prepare($sql);
        return $stmt->execute([
            ':nom'    => $v->getNom(),
            ':prenom' => $v->getPrenom(),
            ':email'  => $v->getEmail(),
            ':tel'    => $v->getTel()
        ]);
    }
}

// ── Contrôleur : VisiteursControleur.php ─────────────────────────
class VisiteursControleur {

    /** Action : enregistrer un nouvel achat */
    public function enregistrer() {
        $visiteur = new Visiteur(
            $_POST['nom'], $_POST['prenom'],
            $_POST['email'], $_POST['tel']
        );
        $dao = new VisiteursDAO();
        $dao->ajouter($visiteur);
        // Redirection vers la vue confirmation
        header("Location: index.php?controleur=confirmation");
    }
}
